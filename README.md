## Pillar Data

    jail_vars:
        jail1:
            name: jail1
            pkg:
              - pkg1
              - pkg2
        jail2:
            name: jail2
            pkg:
              - pkg3
              - pkg4

## State Data pkg

    {% for jailname, jail in pillar.get('jail_vars', {}).items() %}
    {% if jail.pkgs is defined %}
    {{ jailname }}_jail_install_pkgs:
        pkg.installed:
          - jail: {{ jailname }}
          - pkgs:
            # install salt on all jails
            - py36-salt
            {% for packages in jail.pkgs %}
            - {{ packages }}
            {% endfor %}
    {% endif %}
    {% endfor %}
